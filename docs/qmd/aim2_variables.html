This file builds every analysis variable specified in the Aim 2 Statistical Analysis Plan (Tianqi_SAP.docx) from two data sources and writes one row per participant (aim2_master.rds) for the downstream Table 1 / analysis files.

Variables are grouped by data source:

  • REDCap — the longitudinal study database (demographics, comorbidities, kidney function, ICI regimen, concomitant medications).
  • RPDR — the hospital laboratory pull (baseline hemoglobin and albumin).

Each table below lists the variable, its type, and its definition. The R code that creates it is hidden — click ► Code to expand any block.

REDCap is longitudinal. The export has one row per (record_id × redcap_event_name); each variable is pulled from its event, then joined back to one row per record_id:

# Helper: OR across one or more 0/1 checkbox columns → 0/1 (NA treated as 0)
any_box <- function(df, cols) {
  as.integer(rowSums(across(all_of(cols), ~ replace_na(.x == 1, FALSE))) > 0)
}

# Baseline kidney function: egfr at baseline event → KDIGO stage + <60 binary.
# KDIGO bands are inclusive at the top (G2 = 60–89, G1 ≥90); right = FALSE
# with these breaks reproduces that: <15 G5 · 15–29 G4 · 30–44 G3b ·
# 45–59 G3a · 60–89 G2 · ≥90 G1.
kidney <- baseline %>%
  transmute(
    record_id,
    baseline_egfr = as.numeric(egfr),
    baseline_ckd_stage = cut(
      baseline_egfr,
      breaks = c(-Inf, 15, 30, 45, 60, 90, Inf),
      labels = c("G5", "G4", "G3b", "G3a", "G2", "G1"),
      right  = FALSE),
    baseline_ckd_binary = if_else(baseline_egfr < 60, 1L, 0L)  # 1 = G3a–G5
  ) %>%
  mutate(baseline_ckd_stage = factor(
    baseline_ckd_stage, levels = c("G1", "G2", "G3a", "G3b", "G4", "G5")))

# Build a case-insensitive alternation regex per class (terms regex-escaped),
# then flag each class and AIN_con_med_any.
esc <- function(x) gsub("([][(){}.^$*+?\\\\|-])", "\\\\\\1", x)
class_regex <- map_chr(search_terms, ~ paste0(esc(.x), collapse = "|"))

con_text <- prebaseline$con_med %>% replace_na("")

med_flags <- imap_dfc(structured_field, function(field, cls) {
  structured <- if (field %in% names(prebaseline)) {
    replace_na(prebaseline[[field]] == 1, FALSE)
  } else FALSE
  freetext <- str_detect(con_text, regex(class_regex[[cls]], ignore_case = TRUE))
  tibble(!!cls := as.integer(structured | freetext))
})

medications <- bind_cols(record_id = prebaseline$record_id, med_flags) %>%
  mutate(AIN_con_med_any = as.integer(
    rowSums(across(all_of(names(structured_field)))) > 0))

Variable Type Definition
baseline_hgb continuous RPDR hemoglobin closest to baseline visit (±30 d; ties prefer prior)
baseline_albumin continuous RPDR albumin closest to baseline visit (±30 d; ties prefer prior)

Requires an EMPI ↔︎ record_id crosswalk so RPDR (keyed by EMPI) can join to REDCap (keyed by record_id). This block is eval: false until the RPDR pull and crosswalk are available.

# Closest-lab-to-date: smallest |day-diff| from the baseline visit; ties prefer
# the prior value (diff <= 0); anything > 30 days away is coded missing.
closest_lab <- function(lab_df, index_df, group_id, varname, window = 30) {
  lab_df %>%
    filter(Group_Id == group_id, grepl("^\\d+\\.?\\d*$", Result)) %>%
    mutate(Result = as.numeric(Result)) %>%
    inner_join(index_df, by = "EMPI") %>%
    mutate(diff = as.numeric(Seq_Date_Time - baseline_visit_date)) %>%
    filter(abs(diff) <= window) %>%
    arrange(EMPI, abs(diff), diff > 0) %>%
    distinct(EMPI, .keep_all = TRUE) %>%
    transmute(EMPI, !!varname := Result)
}

lab <- read.table(rpdr_lab_path, sep = "|", fill = TRUE, quote = "",
                  header = TRUE) %>%
  select(EMPI, Seq_Date_Time, Result, Group_Id)
lab$Seq_Date_Time <- as.Date(lab$Seq_Date_Time, "%m/%d/%Y")

index_df <- baseline %>%
  transmute(record_id, EMPI, baseline_visit_date = as.Date(visit_date)) %>%
  filter(!is.na(EMPI))

hgb <- closest_lab(lab, index_df, HGB_GROUP, "baseline_hgb")
alb <- closest_lab(lab, index_df, ALB_GROUP, "baseline_albumin")

rpdr_labs <- index_df %>%
  select(record_id, EMPI) %>%
  left_join(hgb, by = "EMPI") %>%
  left_join(alb, by = "EMPI") %>%
  select(record_id, baseline_hgb, baseline_albumin)

# Assemble one row per participant and write aim2_master.rds. RPDR labs are
# attached only if that block was run (data available).
aim2_master <- demographics %>%
  left_join(comorbidities, by = "record_id") %>%
  left_join(autoimmune,    by = "record_id") %>%
  left_join(kidney,        by = "record_id") %>%
  left_join(ici,           by = "record_id") %>%
  left_join(medications,   by = "record_id")

if (exists("rpdr_labs")) {
  aim2_master <- aim2_master %>% left_join(rpdr_labs, by = "record_id")
}

saveRDS(aim2_master, "aim2_master.rds")
glimpse(aim2_master)
